; DOCUMENTATION ONLY — NOT AN EXECUTED TEST.
;
; This file records the expected CST shape for the std.io.formats occurrence
; surface added in task 4284.  It is placed in test/corpus/ to keep CST docs
; co-located with other grammar documentation, but it uses plain comment format
; rather than the `===`/`---` corpus delimiter format precisely so that it does
; NOT look like an executable corpus test.
;
; orchestrator.yaml does not run `tree-sitter test`, so corpus-format files in
; this directory are never validated by CI.  The canonical CI regression signal
; is:
;   - crates/reify-cli/tests/cli_check.rs::check_io_formats_exits_success_no_unresolved
;     (runs the real `reify check` binary on examples/io_formats.ri)
;   - crates/reify-compiler/tests/examples_smoke.rs::all_examples_parse_and_compile_with_stdlib
;     (auto-discovers examples/**/*.ri and fails on any Error-severity diagnostic)
;
; --- enum STEPVersion — three-variant STEP protocol enum ---
;
; Source:
;   enum STEPVersion { AP203, AP214, AP242 }
;
; Expected CST:
;   (source_file
;     (enum_definition
;       name: (identifier)
;       (enum_variant name: (identifier))
;       (enum_variant name: (identifier))
;       (enum_variant name: (identifier))))
;
; --- structure def DisplayStyle — two params with concrete defaults ---
;
; Source:
;   structure def DisplayStyle {
;       param opacity : Real = 1.0
;       param wireframe : Bool = false
;   }
;
; Expected CST:
;   (source_file
;     (structure_definition
;       name: (identifier)
;       (param_declaration
;         name: (identifier)
;         type: (type_expr (identifier))
;         default: (number_literal))
;       (param_declaration
;         name: (identifier)
;         type: (type_expr (identifier))
;         default: (boolean_literal))))
;
; --- occurrence def STLOutput : Output — trait-bound with subject, path, constraint ---
;
; Source:
;   occurrence def STLOutput : Output {
;       param subject : Solid
;       param path : String
;       param resolution : Length = 0.1mm
;       param format : OutputFormat = OutputFormat.STL
;       constraint determined(subject)
;   }
;
; Expected CST:
;   (source_file
;     (occurrence_definition
;       name: (identifier)
;       traits: (trait_bound_list (identifier))
;       (param_declaration
;         name: (identifier)
;         type: (type_expr (identifier)))
;       (param_declaration
;         name: (identifier)
;         type: (type_expr (identifier)))
;       (param_declaration
;         name: (identifier)
;         type: (type_expr (identifier))
;         default: (number_literal))
;       (param_declaration
;         name: (identifier)
;         type: (type_expr (identifier))
;         default: (enum_access
;           enum: (identifier)
;           variant: (identifier)))
;       (constraint_declaration
;         (function_call
;           function: (identifier)
;           (identifier)))))
